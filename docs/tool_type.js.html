/**
 * TypeHelper
 * @namespace
 */
var TypeHelper = {
  /**
   * 判断对象是否是某一数据类型
   * @method
   * @param { string } type 所判断的类型
   * @param { * } obj 待判定对象
   * @return { boolean } 返回判断结果
   * @example
   * is('string', '') => true
   * is('object', {}) => true
   * is('array', []) => true
   * is('function', function(){}) => true
   * is('number', 12) => true
   * is('nan', NaN) => true
   * is('undefined', undefined) => true
   * is('null', null) => true
   */
  is: function (type, obj) {
    type = type.capitalize()
    return Type['is' + type] && Type['is' + type](obj)
  }
}

module.exports = TypeHelper
